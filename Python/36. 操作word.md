- doc 转 docx 需要 pypewin32 库

```
import sys
import pickle
import re

from win32com import client
word = client.Dispatch('Word.Application')
doc = word.Documents.Open(u'E:/word.doc')
doc.SaveAs(u'E:/word.docx', 12, False, "", True, "", False, False, False, False)  # 转化后路径下的文件
doc.Close()
word.Quit()
```
