#### 自定义 装饰器

```python
import time
def timeer(func):
    def inner(*args, **kwargs):
        start = time.time()
        res = func(*args, **kwargs)
        end = time.time()
        print(end - start)
        return res
    return inner

@timeer   # 相当于 test = timmer(test)
def test(n):
    for x in range(n):
        print(x)


```
